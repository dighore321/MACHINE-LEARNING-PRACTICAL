import math
import copy
import numpy as np
import matplotlib
import matplotlib.pyplot as plt
from scipy.special import expit as sigmoid

# Set random seed for reproducibility
np.random.seed(0)

# Generate synthetic dataset
def generate_dataset(N_points):
    # Class 1: Points within radius 0-0.5
    radiuses = np.random.uniform(0, 0.5, size=N_points//2)
    angles = np.random.uniform(0, 2 * math.pi, size=N_points//2)
    x_1 = (radiuses * np.cos(angles)).reshape(N_points//2, 1)
    x_2 = (radiuses * np.sin(angles)).reshape(N_points//2, 1)
    x_class_1 = np.concatenate((x_1, x_2), axis=1)
    y_class_1 = np.full((N_points//2,), 1)
    
    # Class 0: Points within radius 0.6-1
    radiuses = np.random.uniform(0.6, 1, size=N_points//2)
    angles = np.random.uniform(0, 2 * math.pi, size=N_points//2)
    x_1 = (radiuses * np.cos(angles)).reshape(N_points//2, 1)
    x_2 = (radiuses * np.sin(angles)).reshape(N_points//2, 1)
    x_class_0 = np.concatenate((x_1, x_2), axis=1)
    y_class_0 = np.full((N_points//2,), 0)
    
    x = np.concatenate((x_class_1, x_class_0), axis=0)
    y = np.concatenate((y_class_1, y_class_0), axis=0)
    return x, y

# Generate dataset
N_points = 1000
X, y = generate_dataset(N_points)

# Plot the dataset
plt.scatter(X[:N_points//2, 0], X[:N_points//2, 1], color='red', label='class 1')
plt.scatter(X[N_points//2:, 0], X[N_points//2:, 1], color='blue', label='class 0')
plt.legend(loc=9, bbox_to_anchor=(0.5, -0.1), ncol=2)
plt.show()

# Initialize weights
initial_weights = {
    'W1': np.random.rand(3, 2),  # 3 hidden neurons, 2 input features
    'b1': np.zeros(3),           # Bias for hidden layer
    'W2': np.random.rand(3),     # Weights for output layer
    'b2': 0                      # Bias for output
}

# Forward propagation
def forward_propagation(X, weights):
    Z1 = np.dot(X, weights['W1'].T) + weights['b1']
    H = sigmoid(Z1)
    Z2 = np.dot(H, weights['W2']) + weights['b2']
    Y = sigmoid(Z2)
    return Y, Z2, H, Z1

# Backpropagation
def back_propagation(X, Y_T, weights):
    N_points = X.shape[0]
    # Forward propagation
    Y, Z2, H, Z1 = forward_propagation(X, weights)
    
    # Compute loss (binary cross-entropy)
    L = (1 / N_points) * np.sum(-Y_T * np.log(Y + 1e-10) - (1 - Y_T) * np.log(1 - Y + 1e-10))
    
    # Backpropagation
    dLdY = (Y - Y_T) / (Y * (1 - Y) + 1e-10) / N_points
    dLdZ2 = dLdY * Y * (1 - Y)  # Derivative of sigmoid
    dLdW2 = np.dot(dLdZ2, H)
    dLdb2 = np.sum(dLdZ2)
    
    dLdH = np.dot(dLdZ2.reshape(-1, 1), weights['W2'].reshape(1, -1))
    dLdZ1 = dLdH * H * (1 - H)  # Derivative of sigmoid
    dLdW1 = np.dot(dLdZ1.T, X)
    dLdb1 = np.sum(dLdZ1, axis=0)
    
    gradients = {
        'W1': dLdW1,
        'b1': dLdb1,
        'W2': dLdW2,
        'b2': dLdb2
    }
    return L, gradients

# Training loop
def train(X, y, weights, learning_rate=0.1, epochs=1000):
    weights = copy.deepcopy(weights)
    losses = []
    
    for epoch in range(epochs):
        loss, gradients = back_propagation(X, y, weights)
        losses.append(loss)
        
        # Update weights
        weights['W1'] -= learning_rate * gradients['W1']
        weights['b1'] -= learning_rate * gradients['b1']
        weights['W2'] -= learning_rate * gradients['W2']
        weights['b2'] -= learning_rate * gradients['b2']
        
        if epoch % 100 == 0:
            print(f"Epoch {epoch}, Loss: {loss:.4f}")
    
    return weights, losses

# Visualization function
def visualization(weights, X_data, title, superposed_training=False):
    N_test_points = 1000
    xs = np.linspace(1.1 * np.min(X_data), 1.1 * np.max(X_data), N_test_points)
    datapoints = np.transpose([np.tile(xs, len(xs)), np.repeat(xs, len(xs))])
    Y_initial = forward_propagation(datapoints, weights)[0].reshape(N_test_points, N_test_points)
    X1, X2 = np.meshgrid(xs, xs)
    plt.pcolormesh(X1, X2, Y_initial)
    plt.colorbar(label='P(1)')
    
    if superposed_training:
        plt.scatter(X_data[:N_points//2, 0], X_data[:N_points//2, 1], color='red', label='class 1')
        plt.scatter(X_data[N_points//2:, 0], X_data[N_points//2:, 1], color='blue', label='class 0')
        plt.legend()
    
    plt.title(title)
    plt.show()

# Visualize before training
visualization(initial_weights, X, 'Visualization before learning')

# Train the network
trained_weights, losses = train(X, y, initial_weights)

# Visualize after training
visualization(trained_weights, X, 'Visualization after learning', superposed_training=True)
